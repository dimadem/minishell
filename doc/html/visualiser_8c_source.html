15const char *get_token_type_name(t_token_type type)
16{
17 if (type == PHRASE)

63void build_new_prefix(char *new_prefix, char *prefix, int is_left)
64{
65 int i;
66 int j;
67
68 i = 0;
69 j = 0;

72 new_prefix[j] = prefix[i];
73 i++;
74 j++;
75 }
76 if (is_left)

80 new_prefix[j++] = ' ';
