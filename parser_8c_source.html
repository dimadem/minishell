52t_ast *manage_redirs(t_token **tokens, t_ms_data *data)
53{
54 t_ast *command_node;
55 t_ast *redirect_node;
56 t_token *current_token;
57

62 while (current_token && is_redir_node(current_token))

81{
82 t_token *tmp;

85
