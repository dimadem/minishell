24int handle_add_set_shell_variable(t_env **shell_var, char *line)
25{
26 char *key;
27 char *value;
28 char *nq_value;
29
30 key = NULL;
31 value = NULL;
32 nq_value = NULL;
33 if (*shell_var == NULL)
34 add_shell_var_node(shell_var, line);
35 else
36 {
37 key = ft_strcdup(line, '=');
38 value = ft_strchr(line, '=') + 1;
39 nq_value = ft_remove_quotes(value, '\"');

41 free(nq_value);
42 free(key);
