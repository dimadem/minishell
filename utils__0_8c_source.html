32char *ft_find_path(char *cmd, t_env *envp)
33{
34 char **path;
35 char *tmp_slash;
36 char *tmp_full_path;
37 int i;
38
39 path = ft_split(get_env(envp, "PATH"), ':');
40 i = -1;
41 while (path[++i] != NULL)
42 {
43 if (ft_strncmp(cmd, "/", 1) != 0)
44 {
45 tmp_slash = ft_strjoin(path[i], "/");
46 tmp_full_path = ft_strjoin(tmp_slash, cmd);
47 free(tmp_slash);
48 }
49 else
50 tmp_full_path = ft_strdup(cmd);
51 if (access(tmp_full_path, F_OK) == 0)
52 {
53 ft_free_2d_arr(path);
54 return (tmp_full_path);
55 }
56 free(tmp_full_path);
57 }
58 ft_free_2d_arr(path);
59 return (NULL);
60}
